```python
    # Import necessary libraries
    from transformers import pipeline

    # Initialize the few-shot prompting pipeline
    few_shot_prompting = pipeline('few-shot-classification')

    # Define the prompts and labels
    prompts = ["What is the capital of France?", "Who is the president of the United States?", "Where is the Great Wall of China located?"]
    labels = ["capital of France", "president of the United States", "location of the Great Wall of China"]

    # Use the pipeline to classify the prompts
    results = few_shot_prompting(prompts=prompts, labels=labels)

    # Print the results
    for result in results:
        print(f"Prompt: {result['input_text']}")
        print(f"Label: {result['label']}")
        print(f"Score: {result['score']}")
        print("---")
    ```
