```python
    import numpy as np

    # Numerical computation
    a = np.array([1, 2, 3, 4, 5])
    b = np.array([6, 7, 8, 9, 10])

    # Addition
    c = a + b
    print("Addition: ", c)

    # Subtraction
    d = a - b
    print("Subtraction: ", d)

    # Multiplication
    e = a * b
    print("Multiplication: ", e)

    # Division
    f = a / b
    print("Division: ", f)

    # Square root
    g = np.sqrt(a)
    print("Square root of a: ", g)

    # Exponentiation
    h = np.exp(a)
    print("Exponentiation of a: ", h)

    # Logarithm
    i = np.log(a)
    print("Logarithm of a: ", i)

    # Trigonometric functions
    j = np.sin(a)
    print("Sine of a: ", j)
    k = np.cos(a)
    print("Cosine of a: ", k)
    l = np.tan(a)
    print("Tangent of a: ", l)

    # Other mathematical functions
    m = np.max(a)
    print("Maximum of a: ", m)
    n = np.min(a)
    print("Minimum of a: ", n)
    o = np.sum(a)
    print("Sum of a: ", o)
    p = np.mean(a)
    print("Mean of a: ", p)
    ```
