```python
    import matplotlib.pyplot as plt
    import numpy as np

    # Simple data to display in various forms
    x = np.linspace(0, 2 * np.pi, 400)
    y = np.sin(x ** 2)

    plt.figure(figsize=(10, 6))

    # Plotting the data
    plt.plot(x, y, label='Sine Wave')
    plt.xlabel('x (radians)')
    plt.ylabel('y')
    plt.title('Sine Wave')
    plt.legend()

    plt.show()
    ```
