```python
    # Adding Chapter 14
    class Chapter14:
        def __init__(self, title, content):
            self.title = title
            self.content = content

        def display(self):
            print(f"Chapter {self.title}: {self.content}")

    # Creating an instance of Chapter14
    chapter14 = Chapter14("14", "This is Chapter 14 of the book.")

    # Displaying the Chapter14
    chapter14.display()
    ```
