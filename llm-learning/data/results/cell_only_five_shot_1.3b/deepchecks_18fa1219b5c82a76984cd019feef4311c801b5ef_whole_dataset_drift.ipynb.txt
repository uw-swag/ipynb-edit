```python
    # Example 1: Simple Line Plot
    import matplotlib.pyplot as plt
    x = [1, 2, 3, 4, 5]
    y = [1, 4, 9, 16, 25]
    plt.plot(x, y)
    plt.show()

    # Example 2: Bar Plot
    import matplotlib.pyplot as plt
    x = ['A', 'B', 'C', 'D', 'E']
    y = [1, 3, 2, 4, 3]
    plt.bar(x, y)
    plt.show()

    # Example 3: Pie Chart
    import matplotlib.pyplot as plt
    x = ['A', 'B', 'C', 'D', 'E']
    y = [1, 3, 2, 4, 3]
    plt.pie(y, labels=x)
    plt.show()

    # Example 4: Scatter Plot
    import matplotlib.pyplot as plt
    x = [1, 2, 3, 4, 5]
    y = [1, 4, 9, 16, 25]
    plt.scatter(x, y)
    plt.show()

    # Example 5: Histogram
    import matplotlib.pyplot as plt
    x = [1, 2, 3, 4, 5]
    y = [1, 4, 9, 16, 25]
    plt.hist(x, y)
    plt.show()

    # Example 6: Box Plot
    import matplotlib.pyplot as plt
    x = [1, 2, 3, 4, 5]
    y = [1, 4, 9, 16, 25]
    plt.boxplot(x, y)
    plt.show()
    ```
