```python
    import time

    # Start the stopwatch
    start_time = time.time()

    # Do some task here
    time.sleep(2)  # Simulate a task that takes 2 seconds

    # Stop the stopwatch
    end_time = time.time()

    elapsed_time = end_time - start_time

    print(f"Elapsed time: {elapsed_time} seconds")
    ```
